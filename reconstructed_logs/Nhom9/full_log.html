import pandas as pd
import os

def show_training_log(file_path, save_txt_path=None):
    """
    Hm c file history (dng dictionary cha NaN v Infinity)
    """
    
    if not os.path.exists(file_path):
        print(f" Li: Khng tm thy file ti '{file_path}'")
        return

    try:
        with open(file_path, 'r') as f:
            content = f.read().strip()
            
        try:
            # nh ngha t in  Python hiu cc t l trong file
            context = {
                'NaN': np.nan, 
                'nan': np.nan,
                'Infinity': np.inf,
                'inf': np.inf,
                '-Infinity': -np.inf
            }
            
            data_dict = eval(content, {}, context)
            
            df = pd.DataFrame(data_dict)
            
        except Exception as e:
            print(f" Li parse Dictionary: {e}")
            print(" ang th c nh CSV thng thng...")
            df = pd.read_csv(file_path)

        total_epochs = len(df)
        f_out = open(save_txt_path, 'w') if save_txt_path else None
        
        def log(text):
            print(text)
            if f_out: f_out.write(text + "\n")

        log("=" * 60)
        log(f"TRAINING LOG RECONSTRUCTION")
        log(f"Source: {os.path.basename(file_path)}")
        log(f"Total Epochs: {total_epochs}")
        log("=" * 60)
        log("") 

        if total_epochs == 0:
            print(" File rng.")
            return

        for i, row in df.iterrows():
            epoch = i + 1
            
            loss = row.get('loss', float('nan'))
            rse = row.get('rse', float('nan'))
            corr = row.get('corr', float('nan'))
            
            val_loss = row.get('val_loss', float('nan'))
            val_rse = row.get('val_rse', float('nan'))
            val_corr = row.get('val_corr', float('nan'))

            log(f"Epoch {epoch}/{total_epochs}")
            
            # Format chui, x l in ra 'inf' nu l v cc
            log_line = (f"val_loss={val_loss:.6f}, val_rse={val_rse}, val_corr={val_corr}, "
                        f"loss={loss:.6f}, rse={rse:.6f}, corr={corr:.6f}")
            
            log(log_line)
            log("-" * 60) 

        if f_out:
            print(f"\n  lu log ra file: {save_txt_path}")
            f_out.close()
            
    except Exception as e:
        print(f" C li xy ra: {e}")
